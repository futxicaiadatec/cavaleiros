tempo de batalha


function tempoBatalha (casas) {
  let tempo = 0
  for (let i = 0; i < casas.length; i++) {
    const dific = dificuldade[i]
    let somaCosmos = 0
    for (let j = 0; j < casas[i].length; j++) {
      somaCosmos += casas[i][j]
    }
    tempo += dific / somaCosmos
  }
  return tempo
}
