const div = document.getElementById('tabelando')

let table = ''
for (let i = 0; i < 3; i++) {
  table += '<tr>'
  for (let j = 0; j < 3; j++) {
    table += `<td>${i + 1}${j + 1}</td>`
  }
  table += '</tr>'
}

div.innerHTML = '<table>' +  table + '</table>'
